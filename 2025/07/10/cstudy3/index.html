结构体初探

通过下面一段代码，可以基本了解道结构体的定义、初始化、访问。

struct stuff
{
    int number;
    char code[4];
    float cost;
};

int main()
{
    struct stuff monkey = {
        19,
        "123",
        1.414
    };
    char* buffer = (char*)malloc(sizeof(struct stuff) + 50);
    if (!buffer) {
        exit(-1);
    } 
    sprintf(buffer, "number:%d code:%s cose:%f", monkey.number, monkey.code, monkey.cost);
    fputs(buffer, stdout);
    free(buffer); 
    return 0;
}

嵌套结构

在结构中包含另一个结构的用法如下。访问嵌套结构的成员，要使用两次.运算符，其比 &的优先级更高。

#define LEN 20

struct names
{
    char first[LEN];
    char last[LEN];
};

struct guy
{
    struct names handle;
    char favfood[LEN];
    char job[LEN];
    float income;
};

int main(void)
{
    struct guy fellow =
    {
        {"liu","hang" },
        "food",
        "teacher",
        520.2
    };
    char* buffer = (char*)malloc(sizeof(struct guy) + sizeof("name:%s %s, favfood:%s, job: %s, income: %f"));
    if (!buffer) exit(-1);
    sprintf(buffer, "name:%s %s, favfood:%s, job: %s, income: %f", 
        fellow.handle.first, fellow.handle.last, fellow.favfood, fellow.job, fellow.income);

    fputs(buffer, stdout);
    free(buffer);
    return 0;
}

指向结构的指针

用指针访问成员，使用方法例如： p == &gou[0], p->age为gou[0].age或者(*p).age

typedef struct {
    char xing[5];
    char ming[5];
}Name;
typedef struct{
    Name name_val;
    int age;
}Person;

int main(void)
{
    Person gou[3] = {
        {
            {"Li", "Ming"},
            20
        },
        {
            {"Wan", "Wu"},
            25
        }
    };

    Person* p = gou;
    for(int i = 0; i < 2; i++){
        printf("name: %-5s %-5s, age: %d\n", p->name_val.xing, p->name_val.ming, p->age);
        p++;
    }
    return 0;
}

结构的传递

struct funds
{  
    char bank[50];
    double bankfund;
    char save[50];
    double savefund;
};

double sum_0(double, double);
double sum_1(const struct funds*);
double sum_2(struct funds);

int main(void)
{
    struct funds stan = {
        "Bank",
        4032.58,
        "Lucky",
        8543.94
    };
    printf("stan: %.2f。\n", sum_0(stan.bankfund, stan.savefund));
    printf("stan: %.2f。\n", sum_1(&stan));
    printf("stan: %.2f。\n", sum_2(stan));

    return 0;
} 

double sum_0(double x, double y) {
    return(x + y);
}
double sum_1(const struct funds* money) {
    return(money->bankfund + money->savefund);
}
double sum_2(struct funds money) {
    return(money.bankfund + money.savefund);
}

结构与文件

下面是一个简单的图书馆书本目录。

char* s_gets(char* st, int n);

struct book
{
	char title[MAXTITL];
	char author[MAXAUTL];
	float value;
};
 
int main(void) {
	struct book library[MAXBKS];
	int count = 0;
	int index, filecount;
	FILE* pbooks;
	int size = sizeof(struct book);

	if ((pbooks = fopen("book.dat", "ab+")) == NULL) {
		perror("Can't Open");
		exit(-1);
	}
	rewind(pbooks);
	while (count < MAXBKS && fread(&library[count], size, 1, pbooks)) {
		if (count == 0)
			puts("现在的书目录");
		printf("%s by %s: %.2f\n", library[count].title, library[count].author, library[count].value);
		count++;
	}

	filecount = count;
	if (count == MAXBKS) {
		fputs("书满了", stderr);
		exit(2);
	}

	puts("请添加新书的名字");
	puts("按下回车停止输入");
	while (count < MAXBKS && s_gets(library[count].title, MAXTITL) != NULL && library[count].title[0]!='\0') {
		puts("现在输入作者");
		s_gets(library[count].author, MAXAUTL);
		puts("输入价格");
		scanf("%f", &library[count++].value);
		while (getchar() != '\n')
			continue;
		if (count < MAXBKS)
			puts("输入下一个书");
	}

	if (count > 0) {
		puts("下面是最新书目");
		for (index = 0; index < count; index++)
			printf("%s by %s: %.2f\n", library[index].title, library[index].author, library[index].value);
		fwrite(&library[filecount], size, count - filecount, pbooks);
	}
	else
		puts("No Books?");
    
	puts("Byd");
	fclose(pbooks);
	return 0;
	
}

char* s_gets(char* st, int n) {
	char* ret_val;
	char* find;
	ret_val = fgets(st, n, stdin);
	if (ret_val) {
		find = strchr(st, '\n');
		if (find)
			*find = '\0';
		else
			while (getchar() != '\n')
				continue;
	}
	return ret_val;
}

应用案例

typedef union {
	int int_value;
	float float_value;
	char* string_value;
}Data;

typedef enum {
	INT,
	FLOAT,
	STRING
}DataType;

typedef struct {
	DataType type;
	Data data;
}TypeData;


void print_data(const TypeData* print_data);

int main(void) {
	TypeData data1 = { INT,1 };
	TypeData data2 = { FLOAT };
	data2.data.float_value = 2.5;
	TypeData data3 = { STRING };
	data3.data.string_value = (char*)calloc(sizeof("asd") + 1, sizeof(char));
	if (!data3.data.string_value) exit(-1);
	strncpy(data3.data.string_value, "ads", 3);

	print_data(&data1);
	print_data(&data2);
	print_data(&data3);
	return 0;	
}

void print_data(const TypeData* print_data) {
	switch (print_data->type)
	{
	case INT:	
		printf("int_value: %d\n", print_data->data.int_value);
		break;

	case FLOAT:
		printf("float_value: %f\n", print_data->data.float_value);
		break;

	case STRING:
		printf("string_value: %s\n", print_data->data.string_value);
	default:
		break;
	}
}
