文件的打开和关闭

fopen(打开文件)

FILE *fopen(const char *filename, const char *mode)

int main() {
    FILE* fp;
    fp = fopen("./source/test.txt", "w");// 或者fopen(".\\source\\a.text","w");
    fputs("abcdefghijklmn", fp);//写入
    fclose(fp);

    fp = fopen("./source/test.txt", "r");   
    int c;
    while (1) 
    {
        c = fgetc(fp);
        if (feof(fp))
        {
            break;
        }
        printf("%c", c);
    }
    fclose(fp);
    return 1;
}
文件使用方式 含义 如果指定文件不存在
“r”(只读) 为了输入数据，打开一个已经存在的文本文件 出错
“w”(只写) 为了输出数据，打开一个文本文件 建立一个新的文件
“a”(追加) 向文本文件尾添加数据 建立一个新的文件
“rb”(只读) 为了输入数据，打开一个二进制文件 出错
“wb”(只写) 为了输出数据，打开一个二进制文件 建立一个新的文件
“ab”(追加) 向一个二进制文件尾添加数据 出错
“r+”(读写) 为了读和写，打开一个文本文件 出错
“w+”(读写) 为了读和写，建一个新的文件 建立一个新的文件
“a+”(读写) 打开一个文件，在文件尾进行读写 建立一个新的文件
“rb+”(读写) 为了读和写打开一个二进制文件 出错
“wb+”(读写) 为了读和写，新建一个新的二进制文件 建立一个新的文件
“ab+”(读写) 打开一个二进制文件，在文件尾进行读和写 建立一个新的文件

注：当使用“w”，“wb”，“w+”，“wb+”打开文件时会清除文件原本存储的数据。

fclose(关闭文件)

该函数的返回值是int类型的：如果关闭成功就返回0值，否则返回EOF（-1）值。

int main()
{
	//打开
	FILE* pf = fopen("text.txt","w");
	//判断是否打开成功
	if (pf == NULL)
	{
		perror("fopen");
		return 1;
	}
	//关闭
	if (fclose(pf) == EOF)
	{
		//关闭失败
		perror("fclose");
		return 1;
	}
	pf = NULL;
	return 0;
}

rename(重命名)

int rename(const char *old_filename, const char *new_filename)

如果成功，则返回零。如果错误，则返回 -1，并设置 errno(错误信息)。

int main() {
    char oldname[] = "./source/a.text";
    char newname[] = "./source/newfile.txt";
    if (!rename(oldname, newname)) {
        printf("文件重命名成功");
    }
    else {
        printf("错误：不能重命名该文件");
    } 
    return 1;
}

fflush(刷新缓冲)

int fflush(FILE *stream)

  • 如果成功刷新缓冲区，fflush() 返回 0。

  • 如果发生错误，返回 EOF，并且设置错误标识符（ferror）。

  • 如果 stream 为 NULL，则会刷新所有输出流的缓冲区。

  • 如果 stream 是文件指针，则刷新该文件流的输出缓冲区。

feof(末尾判定)

当设置了与流关联的文件结束标识符时，该函数返回一个非零值，否则返回零。意思就是如果读取到文件末尾，则返回一个非零值

if (feof(fp))
{
	break;
}

文件读写

fputc与fgec(单字符存储与读取)

成功运行时返回所存入字符的ASCII码值，否则返回EOF（-1）值。

int main() {
	FILE* pf = fopen("text.txt", "w");
	int i = 'a';
	for (i = 0; i < 26; i++){
		fputc('a' + i, pf);
	}
	//这时候，pf就是字母表：abcde...了
	fclose(pf);
	pf = fopen("text.txt", "r");//此时读取文件要用"r"的方式打开
	while ((i = fgetc(pf)) != EOF){
		printf("%c ", i);
	}
	fclose(pf);
	pf = NULL;
    return 0;
}

fputs与fgets(快速存储与读取)

fgets只能读取一行的数据

char *fgets(char *str, int n, FILE *stream)

如果成功，该函数返回相同的 str 参数。如果到达文件末尾或者没有读取到任何字符，str 的内容保持不变，并返回一个空指针。

如果发生错误，返回一个空指针。 可以使用连续的fgets读取到两行。

int main() {
   
	FILE* fp  ;
	char str[20];
	fp = fopen("file.txt", "w+");
	fputs("abc\ndef\n", fp);

	rewind(fp); // 需要注意，这里不加rewind的话，指针仍指向文件末尾，所以fgets会返回NULL.
	fgets(str, 8, fp);
	fputs(str, stdout);
	fgets(str, 8, fp);
	fputs(str, stdout);

	fclose(fp);
	fp = NULL;
    return 0;
}

fprintf与fscanf(以文本存储与读取)

int fprintf(FILE *stream, const char *format, ...)

int fscanf(FILE *stream, const char *format, ...)

如果成功，该函数返回成功匹配和赋值的个数。如果到达文件末尾或发生读错误，则前者返回一个负数，后者返回 EOF。

注意，一般而言，fscanf遇到空格便会停止对这个的读取。fprnitf本质如同sprintf，把输入转化为字符形式。fscanf则是根据定义，把字符转化为要求的输出格式。

int main() {
	char str1[10];
	char str2;
	int str3;
	float year;
	FILE* fp;

	fp = fopen("file.txt", "w+");
	fprintf(fp, "%s %c %d %.2f", "We", 'a', 2025, 1.53);
	rewind(fp);
	if (fscanf(fp, "%s %c %d %f", str1, &str2, &str3, &year) != 4) {
		perror("scanf_error");	
	}
	printf("Read String1 |%s|\n", str1); // We
	printf("Read String2 |%c|\n", str2); // are
	printf("Read String3 |%d|\n", str3);
	printf("Read Integer |%f|\n", year);
	fclose(fp);
	return(0);
}

fwrite与fread(以二进制存储与读取)

size_t fwrite(const void *ptr, size_t size, size_t nmemb, FILE *stream)

size_t fread(void *ptr, size_t size, size_t nmemb, FILE *stream)

  • ptr – 这是指向要被写入的元素数组的指针。
  • size – 这是要被写入的每个元素的大小，以字节为单位。
  • nmemb – 这是要写入元素的个数，每个元素的大小为 size 字节。
  • stream – 这是指向 FILE 对象的指针，该 FILE 对象指定了一个输出流。
typedef struct {
	int id;
	char name[50];
	float score;
} Student;

int main() {
	FILE* fp = fopen("students.bin", "wb");
	if (!fp) {
		perror("Failed to open file");
		return 1;
	}
	Student s1 = { 1, "Alice", 95.5f };
	Student s2 = { 2, "Bob", 88.0f };

	fwrite(&s1, sizeof(Student), 1, fp);
	fwrite(&s2, sizeof(Student), 1, fp);
	fclose(fp);

	Student read_student;
	fp = fopen("students.bin", "rb");
	if (!fp) {
		perror("Failed to open file");
		return 1;
	}
	while (fread(&read_student, sizeof(Student), 1, fp) == 1) {
		printf("ID: %d, Name: %s, Score: %.2f\n", read_student.id, read_student.name, read_student.score);
	}
    
	fclose(fp); 
	return(0);
}

指针定位

fseek(自定义偏移)

int fseek(FILE *stream, long int offset, int whence)

该函数运行成功，返回零。否则回非零值。

  • stream – 这是指向 FILE 对象的指针，该 FILE 对象标识了流。

  • offset – 这是相对 whence 的偏移量，以字节为单位。

  • whence – 这是表示开始添加偏移 offset 的位置。它一般指定为下列常量之一：

    常量 描述
    SEEK_SET 文件的开头
    SEEK_CUR 文件指针的当前位置
    SEEK_END 文件的末尾 
int main()
{
	FILE* fp = fopen("file.txt", "w+");
	fputs("abcdefg", fp);

	fseek(fp, 2 * sizeof(char), SEEK_SET);
	printf("1. %c", fgetc(fp));

	fseek(fp, 3 * sizeof(char), SEEK_SET);
	printf("1. %c", fgetc(fp));

	return 0;
}

ftell(计算偏移量)

long int ftell(FILE *stream)

成功后，返回位置指示器的当前值。失败时，返回 -1L，并将 errno 设置为系统特定的正值。

//设置偏移量
fseek(pf, 0, SEEK_END);
//计算偏移量
printf("%d", ftell(pf));

rewind(返回开头)

该函数可以将所传入的文件指针设置指向文件初始位置。

文件缓冲区

int main()
{
	//打开
	FILE* pf = fopen("text.txt", "wb");
	if (pf == NULL)
	{
		perror("fopen");
		return 1;
	}
	//存入
	int a = 10000;
	fwrite(&a, sizeof(int), 1, pf);
	printf("此20秒数据在文件缓冲区内，打开文件是没有数据的\n");
	Sleep(20000);//睡眠10秒
	fflush(pf);//此函数可以刷新缓冲区中的数据，使其存入硬盘文件中
	printf("此20秒数据从文件缓冲区内读入到文件中，打开文件是有数据的\n");
	Sleep(20000);
	//关闭
	fclose(pf);
	pf = NULL;
	return 0;
}

因为有缓冲区的存在，C语言在操作文件的时候，需要做刷新缓冲区或者在文件操作结束的时候关闭文件。
如果不做，可能导致读写文件的问题。

这也就是说，如果我们不用fflush，如果我们在一个fopen里面写入、然后读取是会失败的。
